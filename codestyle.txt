import React, { useEffect, useState } from 'react'
import image from '../assets/Homepageimg/landingpageimg/top-view-food-frame-with-copy-space.jpg'

function LandingPage() {
  const fullText = 'Calorie counted for you'
  const [displayedText, setDisplayedText] = useState('')
  const [index, setIndex] = useState(0)
  const [deleting, setDeleting] = useState(false)

  useEffect(() => {
    const timeout = setTimeout(() => {
      if (!deleting) {
        if (index < fullText.length) {
          setDisplayedText(fullText.slice(0, index + 1))
          setIndex(index + 1)
        } else {
          setDeleting(true)
        }
      } else {
        if (index > 0) {
          setDisplayedText(fullText.slice(0, index - 1))
          setIndex(index - 1)
        } else {
          setDeleting(false)
        }
      }
    }, 150)

    return () => clearTimeout(timeout)
  }, [index, deleting, fullText])

  return (
    <div
      className="
    pt-16 
    min-h-screen w-full 
    bg-cover bg-center bg-no-repeat
    sm:pt-20 md:pt-24 lg:pt-28
    sm:min-h-[80vh] md:min-h-[90vh] lg:min-h-screen
    px-4 sm:px-6 md:px-8
  "
      style={{ backgroundImage: `url(${image})` }}
    >
      <div className="flex flex-col items-center md:items-end justify-center h-[calc(100vh-64px)] md:pr-20 space-y-4 text-center md:text-right">
        <h1 className="text-white text-3xl sm:text-4xl md:text-5xl lg:text-6xl font-extrabold leading-tight max-w-lg">
          Deliciousness at your fingertips
        </h1>
        <h2
          className="text-amber-300 text-lg sm:text-xl md:text-2xl font-semibold drop-shadow-md border-r-4 border-white pr-1 animate-pulse inline-block w-[200px] sm:w-[240px] md:w-[280px]"
        >
          {displayedText}
        </h2>

        <button
          className="
            mt-6 sm:mt-8
            px-6 sm:px-8 py-2 sm:py-3
            bg-gradient-to-r from-indigo-500 to-purple-500
            text-white
            font-semibold
            rounded-lg
            shadow-lg
            hover:from-indigo-800 hover:to-purple-800
            focus:outline-none focus:ring-4 focus:ring-green-300
            transition
            duration-300
            ease-in-out
            text-sm sm:text-base md:text-lg
          "
        >
          Launch your meal
        </button>
     
        {/* Balanced meals section */}
        <div className="mt-10 sm:mt-[80px] md:mt-[100px] flex flex-col sm:flex-row gap-3 items-center sm:items-end justify-center sm:justify-end flex-wrap">
          <div className="flex flex-col items-center border border-gray-300 rounded-[5px] p-4 sm:p-6 w-56 sm:w-64 shadow-md bg-fuchsia-200">
            <h3 className="text-lg sm:text-xl font-bold mb-2 text-gray-800">Balanced Meals</h3>
            <h6 className="text-gray-600 text-center text-sm sm:text-base">Eat healthy and tasty</h6>
          </div>
          <div className="flex flex-col items-center border border-gray-300 rounded-[5px] p-4 sm:p-6 w-48 sm:w-54 shadow-md bg-fuchsia-200">
            <h3 className="text-lg sm:text-xl font-bold mb-2 text-gray-800">High Protein</h3>
            <h6 className="text-gray-600 text-center text-sm sm:text-base">To gain weight</h6>
          </div>
          <div className="flex flex-col items-center border border-gray-300 rounded-[5px] p-4 sm:p-6 w-56 sm:w-64 shadow-md bg-fuchsia-200">
            <h3 className="text-lg sm:text-xl font-bold mb-2 text-gray-800">Keto Meals</h3>
            <h6 className="text-gray-600 text-center text-sm sm:text-base">To lose weight</h6>
          </div>
        </div>
      </div>
    </div>
  )
}

export default LandingPage
